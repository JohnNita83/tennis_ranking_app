Activate Venv
.\venv\Scripts\Activate.ps1

Start app
python app.py

workflow to push from dev to main:
# Make sure dev is pushed
git push origin dev

# Switch to main
git checkout main

# Update main with the latest from GitHub
git pull origin main

# Merge dev into main
git merge dev

# Push updated main back to GitHub
git push origin main

# Switch back to dev
git checkout dev

Link to app:
https://tennis-ranking-app.onrender.com/

Restore files to the last commit stage (discard local changes)
git checkout -- .
